function vaiSuVinted() { const nomeObiettivo = document.getElementById('nIn').value.trim(); if(!nomeObiettivo) { alert("Dai un nome all'oggetto prima!"); return; } // Creiamo l'URL di ricerca const searchUrl = `https://www.vinted.it/catalog?search_text=${encodeURIComponent(nomeObiettivo)}`; // Metodo del link invisibile: evita il blocco popup dei browser mobili const a = document.createElement('a'); a.href = searchUrl; a.target = '_blank'; a.rel = 'noopener noreferrer'; document.body.appendChild(a); a.click(); document.body.removeChild(a); }
